## ­ЪћД рЃбрЃћрЃЦрЃюрЃўрЃЎрЃБрЃарЃў рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљ

**рЃЏрЃўрЃќрЃљрЃюрЃў**: Event data-рЃА рЃњрЃљрЃЊрЃљрЃфрЃћрЃЏрЃљ listeners-рЃўрЃАрЃЌрЃЋрЃўрЃА.

public function handle(ReservationStatusChanged $event)
{
    // Admin recipients-рЃћрЃЉрЃўрЃА рЃњрЃљрЃюрЃАрЃљрЃќрЃдрЃЋрЃарЃљ

### 4. Job рЃЎрЃџрЃљрЃАрЃћрЃЉрЃў

#### `SendAdminReservationEmail`
**рЃцрЃљрЃўрЃџрЃў**: `app/Jobs/SendAdminReservationEmail.php`

```php
public function handle()
{
    // Reservation рЃЏрЃЮрЃЊрЃћрЃџрЃўрЃА рЃљрЃ«рЃљрЃџрЃў рЃЕрЃљрЃбрЃЋрЃўрЃарЃЌрЃЋрЃљ serialization-рЃўрЃА рЃерЃћрЃЏрЃЊрЃћрЃњ
    $reservation = Reservation::find($this->reservationId);
    
    // Fresh Mailable instance-рЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ
    $mailableClass = get_class($this->mailable);
    $mailable = new $mailableClass($reservation);
    
    // Email-рЃўрЃА рЃњрЃљрЃњрЃќрЃљрЃЋрЃюрЃљ
    Mail::to($this->to)->send($mailable);
}
```

**рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЋрЃљрЃюрЃў**: Job-рЃћрЃЉрЃў reservation ID-рЃА рЃўрЃюрЃљрЃ«рЃљрЃЋрЃћрЃю рЃюрЃљрЃфрЃЋрЃџрЃљрЃЊ рЃЏрЃЌрЃћрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃАрЃљ serialization рЃърЃарЃЮрЃЉрЃџрЃћрЃЏрЃћрЃЉрЃўрЃА рЃЌрЃљрЃЋрЃўрЃЊрЃљрЃю рЃљрЃАрЃљрЃфрЃўрЃџрЃћрЃЉрЃџрЃљрЃЊ.

### 5. Mail рЃЎрЃџрЃљрЃАрЃћрЃЉрЃў

#### рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў: `ClientConfirmedEmail`
**рЃцрЃљрЃўрЃџрЃў**: `app/Mail/Client/ClientConfirmedEmail.php`

```php
class ClientConfirmedEmail extends Mailable
{
    public $reservation;
    public $restaurantName;

    public function __construct($reservation)
    {
        $this->reservation = $reservation;
        
        // Restaurant name-рЃўрЃА рЃгрЃўрЃюрЃљрЃАрЃгрЃљрЃа рЃњрЃљрЃЏрЃЮрЃЌрЃЋрЃџрЃљ serialization рЃърЃарЃЮрЃЉрЃџрЃћрЃЏрЃћрЃЉрЃўрЃА рЃЌрЃљрЃЋрЃўрЃЊрЃљрЃю рЃљрЃАрЃљрЃфрЃўрЃџрЃћрЃЉрЃџрЃљрЃЊ
        if (method_exists($reservation, 'getRestaurantName')) {
            $this->restaurantName = $reservation->getRestaurantName();
        } else {
            $this->restaurantName = 'N/A';
        }
    }

    public function build()
    {
        return $this->subject('рЃарЃћрЃќрЃћрЃарЃЋрЃљрЃфрЃўрЃўрЃА рЃЊрЃћрЃбрЃљрЃџрЃћрЃЉрЃў - рЃЊрЃљрЃЊрЃљрЃАрЃбрЃБрЃарЃћрЃЉрЃБрЃџрЃўрЃљ')
                    ->view('emails.client.confirmed')
                    ->with([
                        'reservation' => $this->reservation,
                        'restaurantName' => $this->restaurantName
                    ]);
    }
}
```
